is_at_war_with_5_majors = {
	set_variable = { war_with_majors = 0 }
	if = {
		limit = {
			has_war_with = GER
		}
		add_to_variable = {
			var = war_with_majors
			value = 1
		}
	}
	if = {
		limit = {
			has_war_with = FRA
		}
		add_to_variable = {
			var = war_with_majors
			value = 1
		}
	}
	if = {
		limit = {
			has_war_with = ITA
		}
		add_to_variable = {
			var = war_with_majors
			value = 1
		}
	}
	if = {
		limit = {
			has_war_with = ENG
		}
		add_to_variable = {
			var = war_with_majors
			value = 1
		}
	}
	if = {
		limit = {
			has_war_with = SOV
		}
		add_to_variable = {
			var = war_with_majors
			value = 1
		}
	}
	if = {
		limit = {
			has_war_with = USA
		}
		add_to_variable = {
			var = war_with_majors
			value = 1
		}
	}
	if = {
		limit = {
			has_war_with = JAP
		}
		add_to_variable = {
			var = war_with_majors
			value = 1
		}
	}
	if = {
		limit = {
			check_variable = { war_with_majors > 4 }
		}
		set_country_flag = War_with_5_Majors
	}
	
}
