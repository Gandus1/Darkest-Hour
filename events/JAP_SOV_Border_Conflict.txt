#  JAP: Soviet Border Tensions Heighten / SOV: Manchurian Border Tensions Heighten
#########################################################################
country_event = {
	id = JAP_SOV_Border_Conflict.5
	title = {
		text = JAP_SOV_Border_Conflict.5.t.JAP
		trigger = {
			original_tag = JAP
		}
	}
	title = {
		text = JAP_SOV_Border_Conflict.5.t.SOV
		trigger = {
			original_tag = SOV
		}
	}
	desc = JAP_SOV_Border_Conflict.5.d
	picture = GFX_report_event_generic_read_write
	trigger = {
		OR = {
			original_tag = JAP
			original_tag = SOV
		}
		NOT = { has_country_flag = JAP_SOV_Border_Tensions }
		date > 1936.2.4
		has_start_date < 1936.2.4
	}
	mean_time_to_happen = {
		days = 2
	}
	immediate = {
		set_country_flag = JAP_SOV_Border_Tensions
		set_country_flag = JAP_SOV_Turan_Incident
		set_variable = {
			var = JAP_SOV_Balance_of_Power
			value = 50 # 50 is balanced, if it's 40 then it leans towards Japan, if 60 then it leans towards USSR
		}
		clamp_variable = {
			var = JAP_SOV_Balance_of_Power
			min = 0
			max = 100
		}
	}
	#
	option = {
		name = JAP_SOV_Border_Conflict.5.A
		trigger = {
			original_tag = JAP
		}
		unlock_decision_category_tooltip = JAP_Soviet_Japanese_Border_Clashes_category
	}
	#
	option = {
		name = JAP_SOV_Border_Conflict.5.B
		trigger = {
			original_tag = SOV
		}
		unlock_decision_category_tooltip = JAP_Soviet_Japanese_Border_Clashes_category
	}
}
